-- AUTOPARTES VENCES -- PASO 1: Usuarios, ventas e historial sin romper el catálogo público. -- Ejecuta esto en Supabase > SQL Editor > New query > Run. -- Este script solo AGREGA tablas/columnas nuevas. No borra inventario ni fotos. create extension if not exists pgcrypto; -- 1) Perfil visible de cada usuario que entra al admin. create table if not exists public.perfiles ( id uuid primary key references auth.users(id) on delete cascade, email text, nombre text, rol text not null default 'vendedor' check (rol in ('dueno', 'admin', 'vendedor')), activo boolean not null default true, created_at timestamptz not null default now(), updated_at timestamptz not null default now() ); -- 2) Columnas nuevas en piezas para saber quién creó, editó o vendió. alter table public.piezas add column if not exists creado_por uuid references auth.users(id) on delete set null; alter table public.piezas add column if not exists actualizado_por uuid references auth.users(id) on delete set null; alter table public.piezas add column if not exists actualizado_en timestamptz; alter table public.piezas add column if not exists vendido_por uuid references auth.users(id) on delete set null; alter table public.piezas add column if not exists vendido_en timestamptz; alter table public.piezas add column if not exists precio_venta numeric; alter table public.piezas add column if not exists metodo_pago text; alter table public.piezas add column if not exists nota_venta text; -- 3) Registro de ventas. Aunque una pieza se vuelva a publicar, la venta queda guardada aquí. create table if not exists public.ventas ( id uuid primary key default gen_random_uuid(), pieza_id uuid references public.piezas(id) on delete set null, folio text, pieza text, marca text, modelo text, anio text, precio_lista numeric, precio_venta numeric, metodo_pago text, nota text, vendedor_id uuid references auth.users(id) on delete set null, vendedor_email text, vendedor_nombre text, vendido_en timestamptz not null default now(), created_at timestamptz not null default now() ); -- 4) Historial de movimientos importantes del admin. create table if not exists public.movimientos_inventario ( id bigserial primary key, pieza_id uuid, folio text, pieza text, accion text not null, detalle jsonb not null default '{}'::jsonb, usuario_id uuid references auth.users(id) on delete set null, usuario_email text, usuario_nombre text, created_at timestamptz not null default now() ); -- 5) Índices para que estadísticas e historial no se vuelvan lentos. create index if not exists idx_perfiles_email on public.perfiles(email); create index if not exists idx_piezas_vendido_en on public.piezas(vendido_en); create index if not exists idx_piezas_vendido_por on public.piezas(vendido_por); create index if not exists idx_ventas_vendido_en on public.ventas(vendido_en); create index if not exists idx_ventas_vendedor_id on public.ventas(vendedor_id); create index if not exists idx_movimientos_created_at on public.movimientos_inventario(created_at); create index if not exists idx_movimientos_usuario_id on public.movimientos_inventario(usuario_id); -- 6) Seguridad para las tablas nuevas. alter table public.perfiles enable row level security; alter table public.ventas enable row level security; alter table public.movimientos_inventario enable row level security; -- Perfiles: todos los usuarios autenticados pueden ver perfiles, y cada usuario puede crear/actualizar el suyo. drop policy if exists perfiles_select_auth on public.perfiles; create policy perfiles_select_auth on public.perfiles for select to authenticated using (true); drop policy if exists perfiles_insert_self on public.perfiles; create policy perfiles_insert_self on public.perfiles for insert to authenticated with check (auth.uid() = id); drop policy if exists perfiles_update_self on public.perfiles; create policy perfiles_update_self on public.perfiles for update to authenticated using (auth.uid() = id) with check (auth.uid() = id); -- Ventas: usuarios autenticados pueden ver e insertar ventas. drop policy if exists ventas_select_auth on public.ventas; create policy ventas_select_auth on public.ventas for select to authenticated using (true); drop policy if exists ventas_insert_auth on public.ventas; create policy ventas_insert_auth on public.ventas for insert to authenticated with check (auth.uid() = vendedor_id); -- Movimientos: usuarios autenticados pueden ver e insertar historial. drop policy if exists movimientos_select_auth on public.movimientos_inventario; create policy movimientos_select_auth on public.movimientos_inventario for select to authenticated using (true); drop policy if exists movimientos_insert_auth on public.movimientos_inventario; create policy movimientos_insert_auth on public.movimientos_inventario for insert to authenticated with check (auth.uid() = usuario_id);
